<script id="map_vShader" type="x-shader/x-vertex">
      attribute vec4 wPoint;
      attribute float speed;
      
      uniform mat4 mapMatrix;
      varying vec4 aPos;
      varying float aSpeed;
       
      void main() {
		
  		float p_size = 5.+speed/10.;
  	    	   
  		
        gl_Position =  mapMatrix*wPoint;    	
		gl_PointSize = p_size;
		
		aPos = wPoint;
		aSpeed = speed;
 		
      }
    </script>
    
    <script id="map_fShader" type="x-shader/x-fragment">
      precision mediump float;  
 	  varying vec4 aPos;
      varying float aSpeed;
	
 

      void main() {

		vec4 col;				
		col = vec4(aSpeed, 1.-aSpeed, 0.5, 0.2); 		
		gl_FragColor = col;
      }
    </script>
