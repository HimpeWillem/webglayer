<script id="map_vShader" type="x-shader/x-vertex">
      attribute vec4 wPoint;
      attribute float speed;
      
      uniform mat4 attMatrix;
      uniform mat4 mapMatrix;
      uniform mat4 rasterMatrix;
       
      uniform sampler2D uSampler;
       
      varying vec4 aPos;
      varying float aSpeed;
      varying vec4 col;
     
       
      void main() {
		
  		float p_size = 10.;
  	    	   
  		vec4 p =  mapMatrix * wPoint;
  		vec4 rp = rasterMatrix * p;
  		vec4 fdata = texture2D(uSampler, vec2(rp[0],rp[1]));
  		vec4 atr1 = attMatrix*vec4(speed, 0., 0., 1.);;
  		
  		if (fdata[1]>0.){
  			col =  fdata;
  		} else {  		   
  		   col = vec4(speed, 1.-speed, 0.5, 0.2); 	
  		}
  		
        gl_Position = p;    	
		gl_PointSize = p_size;
		
		aPos = wPoint;
		aSpeed = atr1[0];
		
 		
      }
    </script>
    
    <script id="map_fShader" type="x-shader/x-fragment">
      precision mediump float;  
 	  varying vec4 aPos;
      varying float aSpeed;
	  varying vec4 col;
 

      void main() {

		//vec4 col;				
		//col = vec4(aSpeed, 1.-aSpeed, 0.5, 0.2); 		
		gl_FragColor = col;
      }
    </script>
